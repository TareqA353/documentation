import { useState, useEffect } from 'react';

export const RedocAPI = () => {
  const [RedocStandalone, setRedocStandalone] = useState(null);

  useEffect(() => {
    import('redoc').then((Redoc) => {
      setRedocStandalone(() => Redoc.RedocStandalone);
    });
  }, []);

  if (!RedocStandalone) {
    return <div>Loading API documentation...</div>;
  }

  return (
    <div style={{ width: '150%'}}>
    <RedocStandalone
      specUrl="openapi.yaml"
      options={{
        scrollXOffset: 20,
        tryItOutEnabled: true,
        hideDownloadButton: true,
        expandResponses: "all",
        nativeScrollbars: true,
        onlyRequiredInSamples: false,
        pathInMiddlePanel: true,
          requiredPropsFirst: true,
          sortEnumValuesAlphabetically: true,
        jsonSampleExpandLevel: 2,
          theme: {
          spacing: {
            unit: 10,  // Base unit of spacing
            sectionHorizontal: 60,  // Horizontal spacing between sections
            sectionVertical: 50,  // Vertical spacing between sections
          },
          sidebar: {
            width: '250px',  // Adjust the width as needed
          },
          rightPanel: {
            width: '450px', // Width of the right panel
          },
        },
      }}
    />
     </div>
  );
}
